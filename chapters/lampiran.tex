\begin{verbatim}
// Contoh kode program
function calculateAccuracy(predicted, actual) {
    let correct = 0;
    for (let i = 0; i < predicted.length; i++) {
        if (predicted[i] === actual[i]) {
            correct++;
        }
    }
    return (correct / predicted.length) * 100;
}
\end{verbatim}
